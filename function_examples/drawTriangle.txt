#include <Wire.h>
#include <MultiSerial.h>
#include <VGA.h>

// Create a VGA instance to communicate with the shield at address 0x4D
VGA vgaInstance = VGA(0x4D);

void setup() {
  // Initialize the shield to get it ready to communicate with
  vgaInstance.begin();
  
  // Draw a blue triangle with corners at 200,200 150,150 100,200
  vgaInstance.drawTriangle(200, 200, 150, 150, 100, 200, vgaInstance.generateColour(0, 0, 255));
}
